let config = {}; // Loaded from dashboard-config.json


function card(title, itemsHtml) {

  const el = document.createElement('section');

  el.innerHTML = `<h2>${title}</h2>${itemsHtml}`;

  return el;

}

function listItem(label, value, extra) {

  return `<div class="item"><div class="label">${label}</div><div class="value">${value}</div>${extra?`<div class="muted">${extra}</div>`:""}</div>`;

}

function listArray(arr) {

  return `<ul style="margin:6px 0 0 16px;padding:0">${arr.map(x=>`<li>${x}</li>`).join("")}</ul>`;

}


function render() {

  const cards = document.getElementById('cards');

  cards.innerHTML = "";


  // Order: IT + AI + International Politics at top (as requested)

  cards.appendChild(card("IT industry", `${listItem("Highlights", listArray(config.it.highlights))}${listItem("Opportunities", listArray(config.it.opportunities))}`));

  cards.appendChild(card("Artificial intelligence", `${listItem("Breakthroughs", listArray(config.ai.breakthroughs))}${listItem("Adoption", listArray(config.ai.adoption))}`));

  cards.appendChild(card("International politics", `${listItem("Positive developments", listArray(config.international.positives))}${listItem("Cooperation & dialogue", listArray(config.international.dialogues))}`));


  // Local & Ritual

  cards.appendChild(card("Local & ritual timings", `

    ${listItem("Location", config.meta.location)}

    ${listItem("Date", config.meta.date)}

    ${listItem("Rahu Kalam", config.ritual.rahuKalam, config.ritual.note)}

    ${listItem("Yamagandam", config.ritual.yamagandam)}

    <span class="pill warn">Auspicious window: Abhijit</span>

    <span class="pill good">Positive alignment</span>

  `));


  // Current affairs

  cards.appendChild(card("Current affairs (India)", `

    ${listItem("Stability signals", listArray(config.india.stability))}

    ${listItem("Civic progress", listArray(config.india.civic))}

  `));


  // Finance

  cards.appendChild(card("Finance & investments", `

    ${listItem("FD outlook", config.finance.fdOutlook)}

    ${listItem("Dividends", listArray(config.finance.dividends))}

    ${listItem("Strategy", listArray(config.finance.strategy))}

  `));


  // Lifestyle

  cards.appendChild(card("Nutrition & lifestyle (Ayurveda)", `

    ${listItem("Pillars", listArray(config.ayurveda.pillars))}

    ${listItem("Seasonal nourishment", listArray(config.ayurveda.seasonal))}

  `));


  // Entertainment

  cards.appendChild(card("Entertainment & culture (Tollywood + symbolism)", `

    ${listItem("Tollywood", listArray(config.cinema.tollywood))}

    ${listItem("Symbolic reflection", listArray(config.cinema.symbolism))}

  `));


  // Sports

  cards.appendChild(card("Cricket highlights", `

    ${listItem("Wins & milestones", listArray(config.sports.highlights))}

    ${listItem("Upcoming excitement", listArray(config.sports.upcoming))}

  `));


  // Real estate

  cards.appendChild(card("Real estate (Hyderabad)", `

    ${listItem("Market confidence", listArray(config.realEstate.confidence))}

    ${listItem("Opportunities", listArray(config.realEstate.opportunities))}

  `));


  document.getElementById('updatedAt').textContent = "Updated: " + new Date().toLocaleString('en-GB');

}


function refreshData() { render(); }

function toggleCompact() {

  const container = document.querySelector('.container');

  const current = getComputedStyle(container).gridTemplateColumns;

  container.style.gridTemplateColumns = current.includes('minmax') ? '1fr' : 'repeat(auto-fit,minmax(320px,1fr))';

}

function downloadJSON() {

  const blob = new Blob([JSON.stringify(config, null, 2)], { type: "application/json" });

  const a = document.createElement('a');

  a.href = URL.createObjectURL(blob);

  a.download = "dashboard-config.json";

  a.click();

}
